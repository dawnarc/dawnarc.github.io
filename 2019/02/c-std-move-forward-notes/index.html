Examples

std::move with reference
#include <vector>
#include <iostream>

struct A {
    int a[100];
    A() {}
    A(const A& other) {
        std::cout << "copy" << std::endl;
    }
    A(A&& other) {
        std::cout << "move" << std::endl;
    }
};

void foo(const A& a) {
    std::vector<A> vA;
    vA.push_back(std::move(a));
}

void bar(A&& a) {
    std::vector<A> vA;
    vA.push_back(std::move(a));
}

void baz(A& a) {
    std::vector<A> vA;
    vA.push_back(std::move(a));
}

int main() {
    A a;
    foo(a);            // "copy"
    foo(std::move(a)); // "copy"
    bar(std::move(a)); // "move"
    //baz(std::move(a)); // compilation error!!
}
