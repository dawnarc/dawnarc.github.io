How to check if game is running

1st way:

#if WITH_EDITOR
#include "Editor.h"
#endif

bool UVCamBlueprintFunctionLibrary::IsGameRunning()
{
#if WITH_EDITOR
    return (GEditor && GEditor->IsPlaySessionInProgress());
#endif
    return true;
}

Reference:
https://docs.unrealengine.com/4.26/en-US/BlueprintAPI/VirtualCamera_1/IsGameRunning/

2nd way:

static bool IsGameRunning(UObject* Object)
{
    if(!Object)
    {
        return false;
    }
    
    if (auto world = Object->GetWorld())
    {
        return world->WorldType == EWorldType::Game || world->WorldType == EWorldType::PIE;
    }

    return false;
}
